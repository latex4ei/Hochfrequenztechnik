\begin{align*}
C_L &= \frac{(S_{22}-\Delta_S S_{11}^*)^*}{|S_{22}|^2 - |\Delta_S|^2} \\

K &= \frac{1 - |S_{11}|^2 - |S_{22}|^2 + |\Delta_S|^2}{2|S_{12}||S_{21}|}\\
\end{align*}

\begin{align*}
K > 1 \qquad |S_{11}| &< 1 \qquad |S_{22}| < 1 \\
|S_{12}S_{21}| &< 1-|S_{22}|^2 \\
|S_{12}S_{21}| &< 1-|S_{11}|^2 \\
\end{align*}
